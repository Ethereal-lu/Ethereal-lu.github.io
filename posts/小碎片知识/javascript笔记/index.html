javaScript 也是万物皆对象。

数据类型#

ES6开始，所有的局部变量声明都用let，ES5及之前所有的变量声明都用var。

ES6开始支持 “use strict” 即严格检查模式，预防javaScript的随意性导致的一些问题，使用时必须写在javaScript的第一行

数值#

弱类型，无需指定具体类型，只看做变量即可。

let a = 1;
let b = 1.2;
let c = 1e5;

字符串#

字符串是常量，不可变。

单引号或双引号都可以，还可以用反引号写多行文字。特殊字符用 \ 转义。\n \r 等与其他语言一样。

模板字符串（和EL表达式一样） ES6才有

let name = "Reindeer";
let greet = "Hello, ${name}";    // "Hello, Reindeer"

方法：java中String的方法基本都有，且名字和作用都一样。在js中字符串可以使用下标，且length是属性不是方法。

数组#

js中数组的长度可变：通过给arr.length赋值，可改变数组长度。若变大，新扩展单元为undefined；若变小，则截取前半段

同一个数组中可以包含任意的数据类型，如下：

let arr = [1, "hello", 2.5, 'c', true];

方法

arr.length;                        // 数组长度
------------------------------------------------------------------------
arr.indexOf(2.5);                  // 通过元素索引下标
------------------------------------------------------------------------
// 若数组中同时包含 数字1 和字符串"1" ，则他俩不一样
------------------------------------------------------------------------
newArr = arr.slice(2)              // 从下标2开始复制到末尾，返回新数组
newArr = arr.slice(2, 4)           // 从下标2开始复制下标4，返回新数组
------------------------------------------------------------------------
// 作为栈使用
arr.push("Reindeer");              // 向数组尾部压入元素
arr.pop();                         // 从数组尾部弹出元素
arr.unshift("Reindeer");           // 向数组头部压入元素
arr.shift();                       // 从数组头部弹出元素
// 两者结合也可以用作队列
------------------------------------------------------------------------
// 其他常用方法
arr.sort();                        // 排序  [1, 2.5, "c", "hello", true]
arr.reverse();                     // 反转  [true, "hello", "c", 2.5, 1]
arr.concat([3, 4, 5]);             // 拼接，返回新数组，不改变原数组。
arr.join('--')                     // 连接为字符串，"1--2.5--c--hello--true"
let arr = [[1, 2, 3], [4, 5, 6],[7, 8, 9]];  // 多维数组

对象#

js 的对象长得和 python 的字典一样。键都为字符串，值任意

1、定义对象

let person = {
    name: "Reindeer",
    gender: "male",
    age: 18
}

2、给对象属性赋值

person.age = 19;

3、删除属性

delete person.age;

4、添加属性

person.phone = "123456789"

流程控制#

if、else 、while、for 等都与其他语言一样。下面列出特有语法：

  • forEach。向 forEach 中传入一个函数，这个函数将作用于 arr 中的所有元素。
arr.forEach(function(value) {
    console.log(value)
})
  • for … of …
let arr = [2, 3, 6, 1, 8, 4];
for (let value of arr) {              // 这里value是数组中的值
    console.log(value);
}
  • for … in …
let arr = [2, 3, 6, 1, 8, 4];
for (let index in arr) {              // 这里index是数组的下标
    console.log(arr[index]);
}

数据结构#

Map#

let map = new Map([["alice", 18], ["mary", 20], ["reindeer", 19]]);
Map(3) {"alice" => 18, "mary" => 20, "reindeer" => 19}

map.get("alice")        // 通过key获得value
18

map.set("alice", 17)    // 用 set 修改值
Map(3) {"alice" => 17, "mary" => 20, "reindeer" => 19}

map.set("lucy", 18)    // 用 set 添加值
Map(4) {"alice" => 17, "mary" => 20, "reindeer" => 19, "lucy" => 18}

map.delete("alice")    // 删除
true

Set#

无序不重复

let set = new Set([3, 5, 6, 2, 5, 8, 2]);    // 去重
Set(5) {3, 5, 6, 2, 8}

set.has(3)             // 判断是否含有指定值
true

set.add(7)             // 添加
Set(6) {3, 5, 6, 2, 8, 7}

set.delete(2)          // 删除
true

Iterator#

遍历数组用 for … in … 和 for … of … 都可以。

遍历 Map 和 Set 只能用 for … of … ，如下：

for (let x of map) {
    console.log(x)
}

for (let x of set) {
    console.log(x)
}

函数#

函数定义#

由于是弱类型，故和python一样，无需指定返回类型。

// 定义方式一
function add(a, b) {          // 关键字 function 的功能和 python 的 def 一样，指明这是一个函数。
    return a + b;
}
// 定义方式二
let add = function(a, b) {
    return a + b;
}
// 以上两种定义方式效果一样，调用方式也一样

参数处理#

调用函数时，若传入的参数比需要的参数少，或者参数的类型不正确，js会返回NaN；若传入的参数比需要的多，函数只接受前n个他需要的参数，其余参数不理会，但是也不报错。故需要我们手动处理参数。

关键字 arguments 是一个数组，其中存储了调用函数时传入的全部的参数。

关键字 **rest **是一个数组，其中存储了调用函数时传入的多余的参数。

// 若想使用 rest 关键字，定义函数时参数必须这样写，即在所有参数的末尾加上 ", ...rest" 。固定写法
function add(a, b, ...rest) {          
    if (arguments.length < 2) {                              // 传入参数过少
        throw "Incorrect number of parameters!"
    }
    if (typeof a !== "number" || typeof b !== "number") {    // 参数类型不正确
        throw "not a number!";
    }
    console.log(rest);
    return a + b;
}
// 若调用函数时为 add(2, 3, 4, 5, 6, 7);
// 则 arguments === [2, 3, 4, 5, 6, 7]     rest === [4, 5, 6, 7]

变量作用域#

let和var的区别以及const， 一句话就是 var 为函数作用域， let 为块作用域

和java一样，变量选择遵守就近原则，即从自己所在的层级开始查找，如果没有，向上一级查找，直至找到window，若还是没有就报错undefined

所有的全局变量都是 window 对象的属性，只要是定义在函数外的变量就是全局变量，函数本身也属于全局变量。

问题：由于所有的全局变量都会注册为window的属性，则当多个js文件的全局变量都注册到window时，若发生重名，就会产生冲突。

解决方法：每个用户将自己的全局变量都注册到自己的命名空间中，仅将自己的命名空间注册到window。

// 唯一全局变量
var Reindeer = {};

// 其余变量都注册到命名空间Reindeer中
Reindeer.name = "xunlu";
Reindeer.add = function(a, b) {
    return a + b;
}

方法#

正常写法

let MrLu = {
    name: 'xunlu',
    birthDay: 2000,
    // 定义在对象中的函数即为方法
    age: function () {
        let now = new Date().getFullYear();
        return now - this.birthDay;  // 此处必须用this，否则报错。而java可以不用this
    }
}

还可以分开写（就离谱）

function getAge() {
    let now = new Date().getFullYear();
    return now - this.birthDay;
}

let MrLu = {
    name: 'xunlu',
    birthDay: 2000,
    age: getAge
}                  // 调用时 ： MrLu.age()

apply 关键字 可以将一个外部方法指向一个对象，并使用对象中的资源。

function getAge() {
    let now = new Date().getFullYear();
    return now - this.birthDay;
}

let MrLu = {
    name: 'xunlu',
    birthDay: 2000,
    age: getAge
}

// js中所有的函数都有 apply 这个方法
getAge.apply(MrLu, []);  // 将函数 getAge 的 this 指向 MrLu 对象，并传入参数为空（本函数需要的参数）

面向对象编程#

原型 （ ES5及之前使用）【除非迫不得已，不要用这个】

let person = {
    name: "Reindeer",
    gender: "male",
    age: 18
};

let jack = {
    name: "jack"
};

// 将 jack 的原型指向 person ，可以将 person 视为模板。此时 jack 也有了 gender 和 age 属性。
jack.__proto__ = person;

class 继承 （ES6 引入） 本质还是_proto_，但是更人性。

// 定义一个教师类
class Teacher {
    constructor(name) {
        this.name = name;
    }

    teach() {
        alert(this.name + " teacher is teaching the lecture");
    }
}

// 实例化
let yo = new Teacher("yo");
---------------------------------------------------------------------------
// 继承
class MathTeacher extends Teacher {
    constructor(name, discipline) {
        super(name);
        this.discipline = discipline;
    }

    // 独有方法
    great() {
        alert("I am a " + this.discipline + " teacher");
    }

    // 重写方法
    teach() {
        alert(this.name + " teacher is teaching the " + this.discipline);
    }
}

let ya = new MathTeacher("ya", "math");

操作 DOM 对象 【重点】#

DOM（Document Object Model）指 文档对象模型。

浏览器页面就是一个 DOM 树形结构！

获取 DOM 节点#

js 文件必须在需获取的标签下面，否则获取不到，因此一般将 js 的引用放在页面的最底部

document.getElementById()                 // id 名        唯一
document.getElementsByTagName()           // 标签名       返回数组
document.getElementsByClassName()         // 类名         返回数组
// 获取到 DOM 节点后，又可以根据 .children  .firstChild  .lastChild  获取相应子节点

修改 DOM 节点#

let id1 = document.getElementById("id1");
id1.innerText = "123";                     // 修改文本内容。 2、3行代码作用一样，会将原标签下所有内容覆盖
id1.innerHTML = "<strong>123</strong>";    // 修改文本内容，可以解析HTML标签
id1.style.color = "red";                   // 修改 CSS 样式

删除 DOM 节点#

先获取父节点，然后根据父节点删除子节点。不能自己删除自己。

<div id="id1">
    <p id="p1">p1</p>
    <p class="p2">p2</p>
</div>

<script>
    let self = document.getElementById("p1");   // 获取自己
    let father = self.parentElement;            // 获取父节点
    father.removeChild(self);                   // 通过父节点删除自己
</script>

获取到父节点之后，也可以通过father.removeChild(father.children[i]);删除相应子节点。注意：在删除过程中，father.children数组在动态改变，需要注意下标的正确性。

插入 DOM 节点#

移动已存在的节点

<p id="python">python</p>

<div id="list">
    <p id="c">c</p>
    <p id="c++">c++</p>
    <p id="java">java</p>
</div>

<script>
    let python = document.getElementById("python");
    let list = document.getElementById("list");
    list.appendChild(python);    // 将 id 为 python 的标签追加到 list 标签中，且删除原标签
</script>

创建新的节点

let newP = document.createElement("p");    // 新建
newP.setAttribute("id", "newP");           // 设置 id 属性，同样方法也可以设置其他属性
newP.innerHTML = "hello, reindeer";        // innerHTML 自然也可以添加文本
list.appendChild(newP);                    // 追加
---------------------------------------------------
list.insertBefore(newP, python);           // 将 newP 节点插入到 python 节点之前。 （新节点， 参考节点）

操作表单#

表单的目的就是提交信息

获取和修改表单信息#

<form action="#" method="POST">
    <p>
        <span>username: </span>  <input type="text" name="username" id="username">
    </p>

    <p>
        <span>gender: </span>        <!-- radio 表示单选框 -->
        <input type="radio" name="gender" value="man" id="boy"> 男
        <input type="radio" name="gender" value="woman" id="girl"> 女
    </p>
</form>

<script>
    let username = document.getElementById("username");
    let boy = document.getElementById("boy");
    let girl = document.getElementById("girl");
    username.value;                   // 获取文本框中的内容
    username.value = "123";           // 设置文本框中的内容

    boy.checked;                      // 获取当前单选框是否被选中，返回 boolean 值
    girl.checked = true;              // 将当前单选框设置为选中
</script>

提交表单#

<!-- onsubmit 给表单绑定一个检测函数 -->
<form action="https://www.baidu.com/" method="POST" onsubmit="return check()">
    <p>
        <span>username: </span>  <input type="text" name="username" id="username">
    </p>
    <!-- name 用于接收提交的值 -->
    <p>
        <span>password: </span>  <input type="password" id="input-password">
    </p>

    <input type="hidden" name="password" id="md5-password">

    <input type="submit">
</form>

<script>
    function check() {
        let username = document.getElementById("username");
        let password = document.getElementById("input-password");
        let md5password = document.getElementById("md5-password");

        md5password.value = md5(password.value);
    }
</script>
